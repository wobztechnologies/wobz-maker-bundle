```php
<?php

namespace App\Infrastructure\DTO\Enums;

enum OrdersAssemblerEnum: string
{
    case ID = 'id';
    case NAME = 'name';
    case PRICE = 'price';
    case QUANTITY = 'quantity';
    case CREATED_AT = 'createdAt';
    case UPDATED_AT = 'updatedAt';
    
    // etc...
    
    // Here you can add some methods if you want.
}

```
